== Haskell

=== Solution 1

Using `$` operator:

[source,haskell]
----
revGE5 :: String -> String
revGE5 s
  | length s >= 5 = reverse s
  | otherwise     = s

spinWords :: String -> String
spinWords s = unwords $ map revGE5 $ words s
----

=== Solution 2

Just move `revGE5` into a `where` clause:

[source,haskell]
----
spinWords :: String -> String
spinWords s = unwords $ map revGE5 $ words s
  where
    revGE5 :: String -> String
    revGE5 w
      | length w >= 5 = reverse w
      | otherwise     = w
----
